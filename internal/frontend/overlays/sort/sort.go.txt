// Package sort — goclr overlay. A self-contained, reflection-free reimplementation
// of the standard library's sort package: interface-based Sort/Stable/Search plus
// the Ints/Strings/Float64s convenience helpers. The Slice family (which the real
// package implements with internal/reflectlite) is provided by the goclr runtime
// shim and only stubbed here, since goclr lowers every function in a compiled
// package and reflectlite is out of scope.
package sort

// Interface is the standard sort.Interface.
type Interface interface {
	Len() int
	Less(i, j int) bool
	Swap(i, j int)
}

// insertionSort sorts data[a:b] with insertion sort (stable, simple, correct).
func insertionSort(data Interface, a, b int) {
	for i := a + 1; i < b; i++ {
		for j := i; j > a && data.Less(j, j-1); j-- {
			data.Swap(j, j-1)
		}
	}
}

// Sort sorts data in ascending order as determined by the Less method.
func Sort(data Interface) {
	n := data.Len()
	quickSort(data, 0, n)
}

func medianOfThree(data Interface, m1, m0, m2 int) {
	if data.Less(m1, m0) {
		data.Swap(m1, m0)
	}
	if data.Less(m2, m1) {
		data.Swap(m2, m1)
		if data.Less(m1, m0) {
			data.Swap(m1, m0)
		}
	}
}

// quickSort sorts data[a:b]; small ranges fall back to insertion sort.
func quickSort(data Interface, a, b int) {
	for b-a > 12 {
		mid := a + (b-a)/2
		medianOfThree(data, mid, a, b-1)
		pivot := a
		left, right := a+1, b-1
		for {
			for left <= right && data.Less(left, pivot) {
				left++
			}
			for left <= right && data.Less(pivot, right) {
				right--
			}
			if left > right {
				break
			}
			data.Swap(left, right)
			left++
			right--
		}
		data.Swap(pivot, right)
		// Recurse into the smaller side, loop on the larger (bounded stack).
		if right-a < b-(right+1) {
			quickSort(data, a, right)
			a = right + 1
		} else {
			quickSort(data, right+1, b)
			b = right
		}
	}
	insertionSort(data, a, b)
}

// Stable sorts data in ascending order, keeping equal elements in original order.
func Stable(data Interface) {
	insertionSort(data, 0, data.Len())
}

// reverse embeds an Interface, inheriting Len/Swap and flipping Less so the
// embedded data sorts in descending order.
type reverse struct {
	Interface
}

// Less returns the opposite of the embedded implementation's Less.
func (r reverse) Less(i, j int) bool {
	return r.Interface.Less(j, i)
}

// Reverse returns the reverse order for data.
func Reverse(data Interface) Interface {
	return reverse{data}
}

// IsSorted reports whether data is sorted.
func IsSorted(data Interface) bool {
	n := data.Len()
	for i := n - 1; i > 0; i-- {
		if data.Less(i, i-1) {
			return false
		}
	}
	return true
}

// --- Interface adapters for the basic slice types ---
//
// IntSlice/Float64Slice/StringSlice attach sort.Interface to a concrete slice.
// goclr's typed box gives each named slice type a distinct runtime identity, so
// sort.Sort dispatches to the right Len/Less/Swap even when several named-slice
// implementers coexist (e.g. a program's own type alongside sort.StringSlice).

// IntSlice attaches Interface to []int, sorting in increasing order.
type IntSlice []int

func (x IntSlice) Len() int           { return len(x) }
func (x IntSlice) Less(i, j int) bool { return x[i] < x[j] }
func (x IntSlice) Swap(i, j int)      { x[i], x[j] = x[j], x[i] }
func (x IntSlice) Sort()              { Sort(x) }
func (x IntSlice) Search(target int) int {
	return Search(len(x), func(i int) bool { return x[i] >= target })
}

// Float64Slice attaches Interface to []float64, sorting NaNs to the front.
type Float64Slice []float64

func (x Float64Slice) Len() int { return len(x) }
func (x Float64Slice) Less(i, j int) bool {
	return x[i] < x[j] || (isNaN(x[i]) && !isNaN(x[j]))
}
func (x Float64Slice) Swap(i, j int) { x[i], x[j] = x[j], x[i] }
func (x Float64Slice) Sort()         { Sort(x) }

// StringSlice attaches Interface to []string, sorting lexicographically.
type StringSlice []string

func (x StringSlice) Len() int           { return len(x) }
func (x StringSlice) Less(i, j int) bool { return x[i] < x[j] }
func (x StringSlice) Swap(i, j int)      { x[i], x[j] = x[j], x[i] }
func (x StringSlice) Sort()              { Sort(x) }
func (x StringSlice) Search(target string) int {
	return Search(len(x), func(i int) bool { return x[i] >= target })
}

// --- convenience helpers ---
//
// These sort concrete slices directly (a fast path that avoids the interface).

func isNaN(f float64) bool { return f != f }

func Ints(x []int) {
	for i := 1; i < len(x); i++ {
		for j := i; j > 0 && x[j] < x[j-1]; j-- {
			x[j], x[j-1] = x[j-1], x[j]
		}
	}
}

func Float64s(x []float64) {
	for i := 1; i < len(x); i++ {
		for j := i; j > 0 && (x[j] < x[j-1] || (isNaN(x[j]) && !isNaN(x[j-1]))); j-- {
			x[j], x[j-1] = x[j-1], x[j]
		}
	}
}

func Strings(x []string) {
	for i := 1; i < len(x); i++ {
		for j := i; j > 0 && x[j] < x[j-1]; j-- {
			x[j], x[j-1] = x[j-1], x[j]
		}
	}
}

func IntsAreSorted(x []int) bool {
	for i := 1; i < len(x); i++ {
		if x[i] < x[i-1] {
			return false
		}
	}
	return true
}

func Float64sAreSorted(x []float64) bool {
	for i := 1; i < len(x); i++ {
		if x[i] < x[i-1] {
			return false
		}
	}
	return true
}

func StringsAreSorted(x []string) bool {
	for i := 1; i < len(x); i++ {
		if x[i] < x[i-1] {
			return false
		}
	}
	return true
}
