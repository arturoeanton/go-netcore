//go:build go1.20

package bytesconv

// goclr-safe overlay: upstream does a zero-copy string<->[]byte conversion with
// unsafe.String/unsafe.Slice (and StringData/SliceData) that goclr cannot lower.
// The standard language conversions are byte-identical here (they copy, where
// upstream aliases) — correct for gin, which treats the results as read-only views.

// StringToBytes converts a string to a byte slice.
func StringToBytes(s string) []byte {
	return []byte(s)
}

// BytesToString converts a byte slice to a string.
func BytesToString(b []byte) string {
	return string(b)
}
