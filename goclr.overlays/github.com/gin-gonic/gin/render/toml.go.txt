package render

// goclr-safe overlay: cuts the github.com/pelletier/go-toml/v2 import at gin's own
// render seam (go-toml's parser/marshaller relies on a pointer-arithmetic AST goclr
// cannot lower). The render.TOML type is preserved so c.TOML() still compiles;
// rendering returns an error (the validation targets never render TOML).

import (
	"errors"
	"net/http"
)

// TOML contains the given interface object.
type TOML struct {
	Data any
}

var TOMLContentType = []string{"application/toml; charset=utf-8"}

// Render (TOML) is unavailable under goclr (see overlay note above).
func (r TOML) Render(w http.ResponseWriter) error {
	return errors.New("goclr: TOML render is unavailable (go-toml uses a pointer-arithmetic AST)")
}

// WriteContentType (TOML) writes TOML ContentType for response.
func (r TOML) WriteContentType(w http.ResponseWriter) {
	writeContentType(w, TOMLContentType)
}
