package binding

// goclr-safe overlay: cuts the encoding/xml import at gin's binding seam (encoding/xml
// is a large reflection-driven codec not yet provided). The binding type is preserved
// so binding.XML compiles; invoking it returns an error (the validation targets bind
// JSON/form, never XML).

import (
	"errors"
	"net/http"
)

var errXMLUnsupported = errors.New("goclr: XML binding is unavailable")

type xmlBinding struct{}

func (xmlBinding) Name() string { return "xml" }

func (xmlBinding) Bind(req *http.Request, obj any) error { return errXMLUnsupported }

func (xmlBinding) BindBody(body []byte, obj any) error { return errXMLUnsupported }
