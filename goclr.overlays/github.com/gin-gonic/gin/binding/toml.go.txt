package binding

// goclr-safe overlay: the real tomlBinding decodes via
// github.com/pelletier/go-toml/v2, whose parser navigates a contiguous AST by
// unsafe pointer arithmetic (danger.Stride) that goclr cannot lower. Cutting the
// import here — at gin's own binding seam, rather than overlaying the third-party
// library — drops go-toml (and its danger package) from the closure entirely. The
// binding type is preserved so binding.go's `TOML = tomlBinding{}` still compiles;
// invoking it returns an error (the goja/gin validation targets never bind TOML).

import (
	"errors"
	"net/http"
)

var errTOMLUnsupported = errors.New("goclr: TOML binding is unavailable (go-toml uses a pointer-arithmetic AST)")

type tomlBinding struct{}

func (tomlBinding) Name() string { return "toml" }

func (tomlBinding) Bind(req *http.Request, obj any) error { return errTOMLUnsupported }

func (tomlBinding) BindBody(body []byte, obj any) error { return errTOMLUnsupported }
